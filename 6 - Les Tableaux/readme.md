```
    var r1 = Math.random() * 20;
    r1 = Math.floor(r1);
    R1.push(r1);
    var r2 = Math.random() * 20;
    r2 = Math.floor(r2);
    R2.push(r2);
}
//var R3 = R1.concat(R2);
var R3 = R1;
for(var i=0;i<R2.length;i++){
    R3.push(R2[i]);
}

```
